class Animal {

    void eat() {
        System.out.println("Animal is eating");
    }
}

// Subclass (Child)
class Dog extends Animal {

    void bark() {
        System.out.println("Dog is barking");
    }
}

public class Main {
    public static void main(String[] args) {

        Dog d = new Dog();

        d.eat();   // calling superclass method
        d.bark();  // calling subclass method
    }
}
